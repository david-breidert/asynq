127  /// 从队列中取出任务。
128  /// Dequeue a task from the queue.
129  async fn dequeue(&self, queues: &[String]) -> Result<Option<TaskMessage>> {
130    let mut conn = self.get_async_connection().await?;
131
132    if queues.is_empty() {
133      return Ok(None);
134    }
135
136    // Try to dequeue from each queue in priority order
137    for queue in queues {
138      // Build keys according to Go asynq dequeue script format:
139      // KEYS[1] -> asynq:{<qname>}:pending
140      // KEYS[2] -> asynq:{<qname>}:paused
141      // KEYS[3] -> asynq:{<qname>}:active
142      // KEYS[4] -> asynq:{<qname>}:lease
143      let script_keys = vec![
144        keys::pending_key(queue), // KEYS[1]
145        keys::paused_key(queue),  // KEYS[2]
146        keys::active_key(queue),  // KEYS[3]
147        keys::lease_key(queue),   // KEYS[4]
148      ];
149
150      // ARGV[1] -> lease expiration unix time
151      // ARGV[2] -> task key prefix (e.g., "asynq:{qname}:t:")
152      let lease_expiry = Utc::now().timestamp() + 3600; // 1 hour lease
153      let task_key_prefix = keys::task_key_prefix(queue);
154      let args = vec![RedisArg::Int(lease_expiry), RedisArg::Str(task_key_prefix)];
155
156      // The dequeue script returns the task message directly (or nil)
157      let result: Option<Vec<u8>> = self
158        .script_manager
159        .eval_script(&mut conn, "dequeue", &script_keys, &args)
160        .await?;
161
162      if let Some(encoded_msg) = result {
163        // Decode the task message directly from the script result
164        let msg = self.decode_task_message(&encoded_msg)?;
165        return Ok(Some(msg));
166      }
167    }
168
169    Ok(None)
170  }

173  /// Mark a task as done.
174  async fn done(&self, msg: &TaskMessage) -> Result<()> {
175    let mut conn = self.get_async_connection().await?;
176
177    // Generate date string for daily statistics - format: YYYY-MM-DD
178    let now = Utc::now();
179    // Calculate stats expiration (end of next day)
180    let tomorrow = now + chrono::Duration::days(1);
181    let end_of_tomorrow = match tomorrow.date_naive().and_hms_opt(23, 59, 59) {
182      Some(dt) => dt.and_utc(),
183      None => return Err(Error::other("Invalid time for end_of_tomorrow")),
184    };
185    let stats_expiration = end_of_tomorrow.timestamp();
186    // Use regular DONE script for non-unique tasks
187    // KEYS[1] -> asynq:{<qname>}:active
188    // KEYS[2] -> asynq:{<qname>}:lease
189    // KEYS[3] -> asynq:{<qname>}:t:<task_id>
190    // KEYS[4] -> asynq:{<qname>}:processed:<yyyy-mm-dd>
191    // KEYS[5] -> asynq:{<qname>}:processed
192    let mut keys = vec![
193      keys::active_key(&msg.queue),
194      keys::lease_key(&msg.queue),
195      keys::task_key(&msg.queue, &msg.id),
196      keys::processed_key(&msg.queue, &now),
197      keys::processed_total_key(&msg.queue),
198    ];
199    // ARGV[1] -> task ID
200    // ARGV[2] -> stats expiration timestamp
201    // ARGV[3] -> max int64 value
202    let args = vec![
203      RedisArg::Str(msg.id.clone()),
204      RedisArg::Int(stats_expiration),
205      RedisArg::Int(i64::MAX),
206    ];
207    if !msg.unique_key.is_empty() {
208      keys.push(msg.unique_key.clone());
209    }
210    let _result: String = self
211      .script_manager
212      .eval_script(&mut conn, "done", &keys, &args)
213      .await?;
214    Ok(())
215  }

606      ];
607
608      self
609        .script_manager
610        .eval_script(&mut conn, "add_to_group_unique", &keys, &args)
611        .await?
612    };
613
614    match result {
615      1 => Ok(TaskInfo::from_proto(
616        &msg,
617        TaskState::Aggregating,
618        None,
619        None,
620      )),
621      0 => Err(Error::TaskIdConflict),
622      -1 => Err(Error::TaskDuplicate),
623      _ => Err(Error::other("Unexpected script result")),
624    }

815  /// 写入服务器状态 - Go: WriteServerState
816  async fn write_server_state(&self, server_info: &ServerInfo, ttl: Duration) -> Result<()> {
817    let mut conn = self.get_async_connection().await?;
818
819    // 使用 ServerInfo 中的信息构建键
820    // Build keys using information from ServerInfo
821    let server_key =
822      keys::server_info_key(&server_info.host, server_info.pid, &server_info.server_id);
823    let workers_key = keys::workers_key(&server_info.host, server_info.pid, &server_info.server_id);
824
825    // 计算过期时间戳
826    // Calculate expiration timestamp
827    let exp_timestamp = (Utc::now()
828      + chrono::Duration::from_std(ttl).map_err(|e| Error::other(format!("invalid ttl: {e}")))?)
829    .timestamp();
830
831    // 1. 将服务器ID添加到 AllServers ZSET，分数为过期时间戳 (服务器跟踪)
832    // Add the server ID to AllServers ZSET with the expiration timestamp as score (server tracking)
833    let _: () = conn
834      .zadd(keys::ALL_SERVERS, &server_key, exp_timestamp as f64)
835      .await?;
836
837    // 2. 将工作者键添加到 AllWorkers ZSET，分数为过期时间戳 (工作者跟踪)
838    // Add the worker key to AllWorkers ZSET with the expiration timestamp as score (worker tracking)
839    let _: () = conn
840      .zadd(keys::ALL_WORKERS, &workers_key, exp_timestamp as f64)
841      .await?;
842
843    // 3. 使用 Lua 脚本原子性地写入服务器状态
844    // Use Lua script to atomically write server state
845    let server_info_bytes = server_info.encode_to_vec();
846    let keys = vec![server_key, workers_key];
847    let string_args = vec![ttl.as_secs().to_string()];
848    let binary_args = vec![server_info_bytes];
849
850    // ARGV[1] = TTL in seconds
851    // ARGV[2] = server info (encoded protobuf as binary)
852    // ARGV[3+] = worker info (暂时为空，需要支持 WorkerInfo)
853    // ARGV[3+] = worker info (currently empty, needs to support WorkerInfo)
854
855    let _: String = self
856      .script_manager
857      .eval_script_with_binary_args(
858        &mut conn,
859        "write_server_state",
860        &keys,
861        &string_args,
862        &binary_args,
863      )
864      .await?;
865
866    Ok(())
867  }

910  /// 订阅任务取消事件 - Go: CancelationPubSub
911  async fn cancellation_pub_sub(
912    &self,
913  ) -> Result<Box<dyn futures::Stream<Item = Result<String>> + Unpin + Send>> {
914    use futures::StreamExt;
915
916    // 获取 PubSub 连接
917    // Get PubSub connection
918    let mut pubsub = self.get_pubsub().await?;
919
920    // 订阅取消频道
921    // Subscribe to cancellation channel
922    pubsub.subscribe(keys::CANCEL_CHANNEL).await?;
923
924    // 获取消息流并将其转换为拥有所有权的 stream
925    // Get message stream and convert it to an owned stream
926    let message_stream = pubsub.into_on_message();
927
928    // 创建一个流，将 Redis 消息转换为任务 ID 字符串
929    // Create a stream that converts Redis messages to task ID strings
930    let stream = message_stream.filter_map(|msg| async move {
931      match msg.get_payload::<String>() {
932        Ok(task_id) => Some(Ok(task_id)),
933        Err(e) => {
934          tracing::warn!("Failed to parse cancellation message: {}", e);
935          Some(Err(Error::other(format!("Failed to parse message: {e}"))))
936        }
937      }
938    });
939
940    Ok(Box::new(Box::pin(stream)))
941  }
