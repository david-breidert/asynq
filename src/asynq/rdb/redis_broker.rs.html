32  /// 从RedisConnection创建新的Redis经纪人实例
33  /// Create a new Redis broker instance from RedisConnection
34  pub async fn new(conn: RedisConnectionType) -> Result<Self> {
35    match conn {
36      RedisConnectionType::Single {
37        connection_info,
38        #[cfg(feature = "tls")]
39        tls_certs,
40      } => {
41        // If a tls_mode was provided in RedisConnectionType, apply it to the
42        // ConnectionInfo addr so redis::Client will create a TLS connection.
43        // This is a no-op when the `tls` feature is not enabled.
44        let client = {
45          #[cfg(feature = "tls")]
46          {
47            // When tls feature is enabled, if a tls_mode was supplied by the
48            // higher-level RedisConnectionType, prefer building a TLS-enabled
49            // client using the redis crate's TLS helpers. Otherwise fall back
50            // to the normal Client::open.
51            let ci = connection_info.clone();
52            if let Some(tls) = tls_certs {
53              // Attempt to build a client with TLS configuration provided by
54              // callers (they can construct a `redis::TlsCertificates` in examples).
55              // This uses the redis crate API to attach rustls-based cert/key
56              // and optional root CA bytes to the client.
57              Client::build_with_tls(ci, tls)?
58            } else {
59              Client::open(ci)?
60            }
61          }
62          #[cfg(not(feature = "tls"))]
63          {
64            Client::open(connection_info)?
65          }
66        };
67        let mut broker = Self {
68          client: RedisClient::Single(client),
69          script_manager: ScriptManager::default(),
70        };
71        broker.init_scripts().await?;
72        Ok(broker)
73      }
74      #[cfg(feature = "cluster")]
75      RedisConnectionType::Cluster(config) => {
76        // 如果配置了使用 RESP3，创建 push_sender 通道
77        // If RESP3 is configured, create push_sender channel
78        let (push_receiver, cluster_client) = {
79          let (tx, rx) = tokio::sync::mpsc::unbounded_channel();
80          let client = redis::cluster::ClusterClientBuilder::new(config)
81            .use_protocol(redis::ProtocolVersion::RESP3)
82            .push_sender(tx)
83            .build()?;
84          (
85            std::sync::Arc::new(tokio::sync::Mutex::new(Some(rx))),
86            client,
87          )
88        };
89        let mut broker = Self {
90          client: RedisClient::Cluster {
91            client: cluster_client,
92            push_receiver,
93          },
94          script_manager: ScriptManager::default(),
95        };
96        broker.init_scripts().await?;
97        Ok(broker)
98      }
99      #[cfg(feature = "sentinel")]
100      RedisConnectionType::Sentinel {
101        master_name,
102        sentinels,
103        redis_connection_info,
104        #[cfg(feature = "tls")]
105        tls_certs,
106      } => {
107        // Use redis::SentinelClient to resolve the master and get a Client.
108        // See https://docs.rs/redis/latest/redis/sentinel/index.html
109        // Build a SentinelClient and wrap it in Arc<Mutex<>> so we can call its &mut async methods
110        // If any sentinel ConnectionInfo contains a username/password, use that to populate
111        // the SentinelNodeConnectionInfo so the resolved master connection will use auth.
112        let mut node_conn_info: Option<SentinelNodeConnectionInfo> = None;
113        if let Some(conn_info) = redis_connection_info {
114          node_conn_info = Some(SentinelNodeConnectionInfo {
115            #[cfg(feature = "tls")]
116            tls_mode: None, // node_conn_info is used for resolved master; we will pass tls via builder when possible
117            redis_connection_info: Some(conn_info),
118          });
119        }
120        // If TLS certificates are provided, use SentinelClientBuilder to attach them
121        #[cfg(feature = "tls")]
122        let sentinel_client = {
123          if let Some(certs) = tls_certs {
124            // SentinelClientBuilder expects ConnectionAddr values for sentinels
125            let addrs: Vec<redis::ConnectionAddr> = sentinels.iter().map(|ci| ci.addr.clone()).collect();
126            let mut builder = redis::sentinel::SentinelClientBuilder::new(
127              addrs,
128              master_name.clone(),
129              SentinelServerType::Master,
130            )?;
131            builder = builder.set_client_to_redis_certificates(certs);
132            builder.build()?
133          } else {
134            // No certs provided; fall back to simple builder
135            SentinelClient::build(sentinels, master_name.clone(), node_conn_info, SentinelServerType::Master)?
136          }
137        };
138        let client = std::sync::Arc::new(tokio::sync::Mutex::new(sentinel_client));
139
140        let mut broker = Self {
141          client: RedisClient::Sentinel { client },
142          script_manager: ScriptManager::default(),
143        };
144        broker.init_scripts().await?;
145        Ok(broker)
146      }
147    }
148  }

157  /// 获取 PubSub 连接
158  /// Get PubSub connection
159  pub async fn get_pubsub(&self) -> Result<RedisPubSub> {
160    match &self.client {
161      RedisClient::Single(client) => {
162        let pubsub = client.get_async_pubsub().await?;
163        Ok(RedisPubSub::Single(pubsub))
164      }
165      #[cfg(feature = "cluster")]
166      RedisClient::Cluster {
167        client,
168        push_receiver,
169      } => {
170        // 对于集群模式，从 push_receiver 取出接收器并获取连接
171        // For cluster mode, take receiver from push_receiver and get connection
172        let mut receiver_option = push_receiver.lock().await;
173        if let Some(receiver) = receiver_option.take() {
174          // 获取一个集群连接用于订阅
175          // Get a cluster connection for subscription
176          let connection = client.get_async_connection().await?;
177          let cluster_pubsub = ClusterPubSubConnection::from_receiver(receiver, connection);
178          Ok(RedisPubSub::Cluster(cluster_pubsub))
179        } else {
180          Err(Error::other(
181            "Cluster PubSub receiver has already been taken. You can only create one PubSub connection for cluster mode."
182          ))
183        }
184      }
185      #[cfg(feature = "sentinel")]
186      RedisClient::Sentinel { client, .. } => {
187        // Lock the sentinel client and obtain an async pubsub from the resolved master
188        let mut guard = client.lock().await;
189        let master_client = guard.async_get_client().await?;
190        let pubsub = master_client.get_async_pubsub().await?;
191        Ok(RedisPubSub::Sentinel(pubsub))
192      }
193    }
194  }

212  /// 从字节解码任务消息,这里不需要修复任何编码错误
213  /// Decode task message from bytes, no need to fix any encoding errors here
214  pub fn decode_task_message(&self, data: &[u8]) -> Result<TaskMessage> {
215    match TaskMessage::decode(data) {
216      Ok(msg) => Ok(msg),
217      // 不要处理这里的错误，直接返回错误就可以了
218      // Do not handle errors here, just return the error
219      Err(decode_err) => Err(Error::ProtoDecode(decode_err)),
220    }
221  }

223  /// 从Task创建TaskMessage
224  /// Create TaskMessage from Task
225  pub(crate) fn task_to_message(&self, task: &Task) -> TaskMessage {
226    TaskMessage {
227      r#type: task.task_type.clone(),
228      payload: task.payload.clone(),
229      headers: task.headers.clone(),
230      id: task
231        .options
232        .task_id
233        .clone()
234        .unwrap_or(Uuid::new_v4().to_string()),
235      queue: if task.options.queue.clone().is_empty() {
236        DEFAULT_QUEUE_NAME.to_string()
237      } else {
238        task.options.queue.clone()
239      },
240      retry: task.options.max_retry,
241      retried: 0,
242      error_msg: String::new(),
243      last_failed_at: 0,
244      timeout: task
245        .options
246        .timeout
247        .map(|d| d.as_secs() as i64)
248        .unwrap_or(0),
249      deadline: task.options.deadline.map(|d| d.timestamp()).unwrap_or(0),
250      unique_key: String::new(),
251      group_key: task.options.group.clone().unwrap_or_default(),
252      retention: task
253        .options
254        .retention
255        .map(|d| d.as_secs() as i64)
256        .unwrap_or(0),
257      completed_at: 0,
258    }
259  }

261  /// 批量写入 scheduler entries，兼容 Go 版 asynq
262  /// Batch write scheduler entries, compatible with Go version asynq
263  pub async fn write_scheduler_entries(
264    &self,
265    entries: &[SchedulerEntry],
266    scheduler_id: &str,
267    ttl_secs: u64,
268  ) -> Result<()> {
269    let mut args: Vec<RedisArg> = Vec::new();
270    args.push(RedisArg::Int(ttl_secs as i64));
271    for entry in entries {
272      let mut buf = Vec::new();
273      entry
274        .encode(&mut buf)
275        .map_err(|e| Error::other(format!("prost encode error: {e}")))?;
276      args.push(RedisArg::Bytes(buf));
277    }
278    let key = keys::scheduler_entries_key(scheduler_id);
279    let mut conn = self.get_async_connection().await?;
280    let _: () = self
281      .script_manager
282      .eval_script(
283        &mut conn,
284        "write_scheduler_entries",
285        std::slice::from_ref(&key),
286        &args,
287      )
288      .await?;
289    // 新增：ZADD 到全局 ZSET，score 为过期时间戳
290    // New: ZADD to global ZSET, score is the expiration timestamp
291    let zset_key = keys::ALL_SCHEDULERS;
292    let expire_at = chrono::Utc::now().timestamp() + ttl_secs as i64;
293    let _: () = conn.zadd(zset_key, &key, expire_at).await?;
294
295    Ok(())
296  }

298  /// 记录调度事件，兼容 Go 版 asynq
299  /// Record scheduling event, compatible with Go version asynq
300  pub async fn record_scheduler_enqueue_event(
301    &self,
302    event: &SchedulerEnqueueEvent,
303    entry_id: &str,
304  ) -> Result<()> {
305    let mut buf = Vec::new();
306    event
307      .encode(&mut buf)
308      .map_err(|e| Error::other(format!("prost encode error: {e}")))?;
309    let key = keys::scheduler_history_key(entry_id);
310    let mut conn = self.get_async_connection().await?;
311    let args = vec![
312      RedisArg::Int(event.enqueue_time.map(|x| x.seconds).unwrap_or(0)),
313      RedisArg::Bytes(buf),
314      RedisArg::Int(1000),
315    ];
316    self
317      .script_manager
318      .eval_script::<()>(&mut conn, "record_scheduler_enqueue_event", &[key], &args)
319      .await?;
320    Ok(())
321  }

323  /// 通过脚本获取所有 SchedulerEntry，兼容 Go 版 asynq
324  /// Get all SchedulerEntry through script, compatible with Go version asynq
325  pub async fn scheduler_entries_script(
326    &self,
327    scheduler_id: &str,
328  ) -> Result<std::collections::HashMap<String, Vec<u8>>> {
329    let key = keys::scheduler_entries_key(scheduler_id);
330    let mut conn = self.get_async_connection().await?;
331    // 调用脚本，假设脚本返回 [key1, value1, key2, value2, ...]
332    // Call the script, assuming the script returns [key1, value1, key2, value2, ...]
333    let result: Vec<Vec<u8>> = self
334      .script_manager
335      .eval_script(&mut conn, "get_scheduler_entries", &[key], &[])
336      .await?;
337    let mut map = std::collections::HashMap::new();
338    let mut iter = result.chunks_exact(2);
339    while let Some([k, v]) = iter.next() {
340      let key_str = String::from_utf8_lossy(k).to_string();
341      map.insert(key_str, v.clone());
342    }
343    Ok(map)
344  }

359  /// 删除 scheduler entries 数据，兼容 Go 版 asynq
360  /// Delete scheduler entries data, compatible with Go version asynq
361  pub async fn clear_scheduler_entries(&self, scheduler_id: &str) -> Result<()> {
362    let key = keys::scheduler_entries_key(scheduler_id);
363    let zset_key = keys::ALL_SCHEDULERS;
364    let mut conn = self.get_async_connection().await?;
365    // ZREM 全局调度器 ZSET
366    // ZREM global scheduler ZSET
367    let _: () = conn
368      .zrem(zset_key, &key)
369      .await
370      .map_err(|e| Error::other(format!("redis ZREM error: {e}")))?;
371    // DEL 具体 entry 列表
372    // DEL specific entry list
373    let _: () = conn
374      .del(&key)
375      .await
376      .map_err(|e| Error::other(format!("redis DEL error: {e}")))?;
377    Ok(())
378  }
